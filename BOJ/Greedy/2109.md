package algo;


import java.io.IOException;
import java.util.*;


public class Main {


    public static void main(String[] args) throws IOException {
        Scanner in = new Scanner(System.in);
        int n = in.nextInt(); // n개의 대학
        // 돈을 최대로 벌 수 있도록 하자
        // 매 순간 최적의 선택


        // 1일안에 30   2일안에 40 3일안에 50 3일안에 60 인 상황이 있다고 해보자

        // 그럼 1 - 40
        // 2- 50
        // 3- 60 짜리를 가는게 최선의 방법

        PriorityQueue<Class> q = new PriorityQueue<>();
        int max = 0;
        for (int i = 0; i < n; i++) {
            int p = in.nextInt();
            int d = in.nextInt();
            if (max < d) {
                max = d;
            }
            Class aClass = new Class(d, p);
            q.add(aClass);
        }
        
        Class[] attend = new Class[max+1];
        while (!q.isEmpty()) {
            Class poll = q.poll();
            for (int i = poll.d; i >= 1; i--) {
                if (attend[i] == null) {
                    attend[i] = poll;
                    break;
                }
            }
        }


        int sum = 0;
        for (int i = 1; i < max + 1; i++) {
            if (attend[i] != null) {
                sum += attend[i].p;
            }
        }
        System.out.println(sum);
    }
}

// 어디에 우선순위를 둬야할까
//가격?
//일자?
// 둘다라면? 어떤 기준으로?
class Class implements Comparable<Class>{
    int d; // 일자
    int p; // 가격

    Class(int d, int p) {
        this.d = d;
        this.p = p;
    }

    @Override
    public int compareTo(Class o) {
        if (this.p < o.p) {
            return 1;
        } else if (this.p == o.p) {
            if (this.d > o.d) {
                return 1;
            }
        }
        return -1;
    }
}
