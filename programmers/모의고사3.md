package algo;

class Solution {
    public int solution(int a, int b, int n) {
        int total = 0;
        int store = 0;
        while (true) {
            System.out.println(total);
            if (n < a) {
                break;
            }

            total += ((n / a) * b);
            store += (n % a);
            n = (n / a) * b;

            if (n < a) {
                n += store;
                store = 0;
            }
        }


        return total;
    }
}
