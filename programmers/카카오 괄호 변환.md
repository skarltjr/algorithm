package algo;

import java.util.Stack;

class Solution {
    public String solution(String p) {
        if (p.isEmpty()) {
            return "";
        }
        if (checkRight(p)) {
            return p;
        }
        StringBuilder result = new StringBuilder();
        //setting
        separateUV(p,result);


        return result.toString();
    }

    private void separateUV(String p, StringBuilder result) {
        StringBuilder u = new StringBuilder();
        StringBuilder v = new StringBuilder();
        if (p.isEmpty()) {
            return;
        }

        int leftCount = 0; // (
        int rightCount = 0; // )
        int lastIndex = 0;
        for (int i = 0; i < p.length(); i++) {
            char target = p.charAt(i);
            lastIndex = i;
            if (target == '(') {
                leftCount++;
                u.append(target);
            } else {
                rightCount++;
                u.append(target);
            }

            if (leftCount == rightCount) {
                break;
            }
        }
        v.append(p.substring(lastIndex+1));


        if (checkRight(u.toString())) {
            // 올바르다면
            result.append(u);
            String target = v.toString();
            separateUV(target, result);
        } else {
            //올바르지 않다면
            StringBuilder temp = new StringBuilder();
            temp.append('(');
            separateUV(v.toString(), temp);
            temp.append(')');
            u.delete(0, 1);
            u.delete(u.length() - 1, u.length());

            //괄호 뒤집어서 붙힌다.
            reversing(temp,u);
            result.append(temp);
        }
    }

    private void reversing(StringBuilder temp, StringBuilder u) {
        for (int i = 0; i < u.length(); i++) {
            temp.append(u.charAt(i) == '(' ? ')' : '(');
        }
    }


    private boolean checkRight(String u) {
        Stack<Character> stack = new Stack<>();
        for (int i = 0; i < u.length(); i++) {
            if (u.charAt(i) == '(') {
                stack.push(u.charAt(i));
            } else {
                if (stack.size() == 0) {
                    return false;
                }
                Character pop = stack.pop();
                if (pop == u.charAt(i)) {
                    return false;
                }
            }
        }
        return true;
    }
}
