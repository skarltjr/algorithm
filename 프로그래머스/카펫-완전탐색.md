```
package algo;


/**
 * 가로 >= 세로
 * (yellow의 가로 + 2 ) * (yellow의 세로 +2 ) = 전체 격자 개수
 * 반대로 생각해보면
 * (전체 가로 - 2) * (전체 세로 - 2) = 노랑개수
 * */
class Solution {
    public int[] solution(int brown, int yellow) {
        int[] answer = {};

        int target = brown + yellow;

        // 약수를 다 구해보는데
        for (int i = 1; i <= target; i++) {
            int x = i; // 가로
            int y = target / x; // 세로

            if (x >= y && (x - 2) * (y - 2) == yellow) {
                return new int[]{x, y};
            }
        }

        return answer;
    }
}
```
