import java.util.HashMap;
import java.util.Locale;
import java.util.Map;

class Solution {

    public String solution(String word) {
        String result = "";

        Map<Character, String> map = new HashMap<>();
        map.put('A', "Z");
        map.put('B', "Y");
        map.put('C', "X");
        map.put('D', "W");
        map.put('E', "V");
        map.put('F', "U");
        map.put('G', "T");
        map.put('H', "S");
        map.put('I', "R");
        map.put('J', "Q");
        map.put('K', "P");
        map.put('L', "O");
        map.put('M', "N");
        map.put('N', "M");
        map.put('O', "L");
        map.put('P', "K");
        map.put('Q', "J");
        map.put('R', "I");
        map.put('S',"H");
        map.put('T',"G");
        map.put('U',"F");
        map.put('V',"E");
        map.put('W',"D");
        map.put('X',"C");
        map.put('Y',"B");
        map.put('Z',"A");



        for (int i = 0; i < word.length(); i++) {
            char ch = word.charAt(i);
            if (ch == ' ') {
                result += " ";
                continue;
            }

            if (Character.isUpperCase(ch)) {
                result += map.get(ch);
                continue;
            } else {
                ch = Character.toUpperCase(ch);
                result += map.get(ch).toLowerCase(Locale.ROOT);
                continue;
            }
        }


        return result;
    }
}
